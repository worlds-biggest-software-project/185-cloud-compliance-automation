```python
import uuid
from datetime import datetime

from sqlalchemy import DateTime, func
from sqlalchemy.dialects.postgresql import UUID
from sqlalchemy.ext.asyncio import AsyncAttrs
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column


class Base(AsyncAttrs, DeclarativeBase):
    pass


class TimestampMixin:
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now(), nullable=False
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now(), onupdate=func.now(), nullable=False
    )


class UUIDPrimaryKeyMixin:
    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True), primary_key=True, default=uuid.uuid4
    )
```

```python
# src/database.py
from collections.abc import AsyncGenerator

from sqlalchemy.ext.asyncio import (
    AsyncSession,
    async_sessionmaker,
    create_async_engine,
)

from src.config import settings

engine = create_async_engine(
    str(settings.database_url),
    pool_size=settings.database_pool_size,
    max_overflow=settings.database_max_overflow,
    echo=settings.debug,
)

async_session_factory = async_sessionmaker(
    engine, class_=AsyncSession, expire_on_commit=False
)


async def get_db_session() -> AsyncGenerator[AsyncSession, None]:
    async with async_session_factory() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
```

#### 1.4 — FastAPI Application Factory and Health Check

**What**: Create the FastAPI app with middleware, CORS, error handling, and a health endpoint.

**Design**:

```python
# src/main.py
from contextlib import asynccontextmanager
from collections.abc import AsyncGenerator

from fastapi import FastAPI, Request
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import JSONResponse

from src.config import settings
from src.database import engine


@asynccontextmanager
async def lifespan(app: FastAPI) -> AsyncGenerator[None, None]:
    # Startup: verify database connectivity
    async with engine.begin() as conn:
        await conn.execute(text("SELECT 1"))
    yield
    # Shutdown: dispose engine
    await engine.dispose()


def create_app() -> FastAPI:
    app = FastAPI(
        title=settings.app_name,
        version=settings.app_version,
        lifespan=lifespan,
        docs_url=f"{settings.api_prefix}/docs",
        openapi_url=f"{settings.api_prefix}/openapi.json",
    )

    app.add_middleware(
        CORSMiddleware,
        allow_origins=["http://localhost:3000"],
        allow_credentials=True,
        allow_methods=["*"],
        allow_headers=["*"],
    )

    @app.get("/health")
    async def health_check() -> dict[str, str]:
        return {"status": "healthy", "version": settings.app_version}

    return app


app = create_app()
```

```python
class ProblemDetail(BaseModel):
```

#### 2.1 — Database Schema (Hybrid Relational + JSONB)

**What**: Implement all SQLAlchemy ORM models matching the hybrid relational+JSONB data model and generate Alembic migrations.

**Design**:

Adopting data-model-suggestion-3 (Hybrid Relational + JSONB) because it balances relational integrity with framework-specific extensibility, uses 16 tables (vs 25+ normalized), and supports OSCAL JSON storage natively.

```python
# src/models/organization.py
class Organization(Base, UUIDPrimaryKeyMixin, TimestampMixin):
    __tablename__ = "organizations"

    name: Mapped[str] = mapped_column(nullable=False)
    slug: Mapped[str] = mapped_column(unique=True, nullable=False)
    subscription_tier: Mapped[str] = mapped_column(default="free", nullable=False)
    settings: Mapped[dict] = mapped_column(JSONB, default=dict, nullable=False)
    # settings schema: {
    #   "industry": str,
    #   "primary_jurisdiction": str (ISO 3166-1),
    #   "jurisdictions": list[str],
    #   "scan_schedule_cron": str,
    #   "notification_preferences": {
    #     "slack_webhook": str | null,
    #     "email_alerts": bool,
    #     "alert_severities": list[str]
    #   }
    # }


class User(Base, UUIDPrimaryKeyMixin, TimestampMixin):
    __tablename__ = "users"
```

```python
# src/services/ai_service.py
from anthropic import Anthropic

class AIService:
    def __init__(self, anthropic_client: Anthropic):
        self.client = anthropic_client

    def _build_control_context(self, frameworks: list[Framework], controls: list[Control]) -> str:
        """Serialize the control library into a context block for prompt caching.
        This context block is cached across requests to avoid re-sending ~100KB
        of control definitions on every call.
        """
        ...

    async def analyze_gap(
        self, org_id: uuid.UUID, description: str, active_frameworks: list[str]
    ) -> GapAnalysisResult:
        """Natural-language gap analysis:
        User describes a system change; AI identifies affected controls.

        System prompt (cached): Full control library for active frameworks
        User prompt: "We are adding a new S3 bucket for customer PII..."
        Response: List of affected controls with risk assessment
        """
        messages = [
            {
                "role": "user",
                "content": description,
            }
        ]
        response = self.client.messages.create(
            model=settings.anthropic_model,
            max_tokens=settings.ai_max_tokens,
            system=[
                {
                    "type": "text",
                    "text": self._build_control_context(frameworks, controls),
                    "cache_control": {"type": "ephemeral"},
                },
                {
                    "type": "text",
                    "text": GAP_ANALYSIS_SYSTEM_PROMPT,
                },
            ],
            messages=messages,
        )
        return self._parse_gap_analysis(response)

    async def generate_remediation(
        self, finding: Finding, resource: CloudResource
    ) -> RemediationPlaybook:
        """Generate step-by-step remediation for a specific finding.
        Includes infrastructure-specific fix instructions and Terraform/CLI commands.
        """
        ...

    ) -> str:
        """Generate a compliance policy document (e.g., access control policy)
        based on organization context and control requirements.
        """
        ...

    async def classify_evidence(
        self, evidence_content: str, available_controls: list[Control]
    ) -> list[EvidenceClassification]:
```
